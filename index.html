    <script>

        const totalSinInteresElem = document.getElementById('totalSinInteres');


        function formatCurrency(amount) {

            return new Intl.NumberFormat('es-AR', { style: 'currency', currency: 'ARS', minimumFractionDigits: 2 }).format(amount);

        }


        function calculate() {

            const amountInput = document.getElementById('amount');

            const amount = parseFloat(amountInput.value);


            if (isNaN(amount) || amount <= 0) {

                amountInput.value = '';

                document.getElementById('results').classList.remove('show');

                return;

            }


            const totalCon5Interes = amount * 1.05;

            const cuotaCon5Interes = totalCon5Interes / 3;

            const totalCon15Interes = amount * 1.15;

            const cuotaCon15Interes = totalCon15Interes / 6;


            const diferencia3Total = totalCon5Interes - amount;

            const diferencia6Total = totalCon15Interes - amount;

            

            const costoExtra3Cuotas = diferencia3Total / 3;

            const costoExtra6Cuotas = diferencia6Total / 6;


            totalSinInteresElem.textContent = formatCurrency(amount);

            document.getElementById('totalCon5Interes').textContent = formatCurrency(totalCon5Interes);

            document.getElementById('cuotaCon5Interes').textContent = formatCurrency(cuotaCon5Interes);

            document.getElementById('totalCon15Interes').textContent = formatCurrency(totalCon15Interes);

            document.getElementById('cuotaCon15Interes').textContent = formatCurrency(cuotaCon15Interes);

            document.getElementById('diferencia3Total').textContent = formatCurrency(diferencia3Total);

            document.getElementById('diferencia6Total').textContent = formatCurrency(diferencia6Total);

            

            document.getElementById('costoExtra3Cuotas').textContent = formatCurrency(costoExtra3Cuotas);

            document.getElementById('costoExtra6Cuotas').textContent = formatCurrency(costoExtra6Cuotas);

            

            document.getElementById('results').classList.add('show');

        }

    </script>
